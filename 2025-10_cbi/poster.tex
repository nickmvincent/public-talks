\begin{minipage}[t]{0.48\textwidth}
\begin{center}
{\fontsize{32}{40}\selectfont\bfseries\color{accent} The Problem}
\end{center}

\vspace{0.3in}

{\fontsize{20}{26}\selectfont\color{primary}

\textbf{Information markets fail:}

\begin{itemize}[leftmargin=0.4in,itemsep=0.15in]
\item Information is cheap to copy, hard to exclude
\item Competitive markets drive price to zero
\item Producers struggle to capture value
\end{itemize}

\vspace{0.3in}

\textbf{AI makes it worse:}

\begin{itemize}[leftmargin=0.4in,itemsep=0.15in]
\item Models extract value more efficiently
\item Erodes traditional protections
\item \textbf{Risk: extreme power concentration}
\end{itemize}
}

\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\begin{center}
{\fontsize{32}{40}\selectfont\bfseries\color{accent} The Solution}
\end{center}

\vspace{0.3in}

{\fontsize{20}{26}\selectfont\color{primary}

\textbf{CBI creates friction to:}

\begin{itemize}[leftmargin=0.4in,itemsep=0.15in]
\item Prevent unchecked value extraction
\item Maintain incentives for quality information
\item Distribute power more evenly
\end{itemize}

\vspace{0.3in}

\textbf{Benefits:}

\begin{itemize}[leftmargin=0.4in,itemsep=0.15in]
\item Higher quality data â†’ better AI
\item Prevents ecosystem collapse
\item Built-in accountability
\end{itemize}
}

\end{minipage}
